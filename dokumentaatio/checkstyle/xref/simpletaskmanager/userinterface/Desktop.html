229     private HBox tabInfo(String header, String imgName) {
230         HBox content = new HBox();
231         Label label = new Label(header.substring(0,4));
232         //label.setAlignment(Pos.CENTER_RIGHT);
233         try {
234             ImageView icon = buildImage(Desktop.class.getResource("../../icons/" + imgName + ".png").toExternalForm());
235             content.getChildren().addAll(icon, label);
236         }
237         catch (Exception e) {
238             content.getChildren().addAll(label);
239         }
240         content.setAlignment(Pos.CENTER);
241         content.setSpacing(3);
242         return content ;
243     }

250     // Helper method to create image from image patch
251     private static ImageView buildImage(String imgPatch) {
252         try {
253             Image i = new Image(imgPatch);
254             ImageView imageView = new ImageView(i);
255             //You can set width and height
256             imageView.setFitHeight(15);
257             imageView.setFitWidth(15);
258             return imageView;
259         } catch (Exception e) {
260             // jotain meni pieleen kuvankasittelyssa. laitetaan vain tyhja.
261             System.out.println("ikonia ei loytynyt");
262             System.out.println(imgPatch);
263             System.out.println(e);
264             return new ImageView();
265         }
266     }
